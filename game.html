What is an Interactive Game?

In addition to creating DIY photoshops, the Processing software can also be used for making interactive games based on a variety of subjects. These games can be as complex as a run-'n-gun game, or as simple as a "catch the ball" game. When people make their own games, they can be used for a variety of purposes (educational, for fun or recreation, etc.). I created this little game in April 2025, which is about catching the seven chakras, which you can learn more about below. Have fun playing this simple, silly game!

Make sure to read the code down below too!

If you want to learn more about the 7 chakras, you can go to any of these sources below!


// Catch the Seven Chakras

var ballx = 300;
var bally = 300;
var ballSize = 60;
var score = 0;
var gameState = "L1";

function setup() {
  createCanvas(600, 600);
  textAlign(CENTER);
  textSize (20);
} // end of set up


function draw() {
  if(gameState == "L1"){
    levelOne();
  }
  if(gameState == "L2"){
    levelTwo();
  }
  if(gameState == "L3"){
    levelThree();
  }
  if(gameState == "L4"){
    levelFour();
  }
  if(gameState == "L5"){
    levelFive();
  }
  if(gameState == "L6"){
    levelSix();
  }
  if(gameState == "L7"){
    levelSeven();
  }

text(("Score: " + score), width/2, 60);
} // end of draw

function levelOne(){
  background (0, 153, 51);
  text("Level 1: The Root", width/2, height-20);
  var distToBall = dist(ballx, bally, mouseX, mouseY);
  if(distToBall < ballSize/2){
    ballx = random(width);
    bally = random(height);
    score = score + 1;
  }
  if(score>=7){
    gameState = "L2";
  }
  
  fill(255, 0, 0);
  ellipse(ballx, bally, ballSize, ballSize);
  line(ballx, bally, mouseX, mouseY);
  
} // end of level 1
